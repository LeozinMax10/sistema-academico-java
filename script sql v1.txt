CREATE DATABASE db_cadastro_alunos;

CREATE TABLE aluno(
	rgm INTEGER PRIMARY KEY,
	nome VARCHAR(50),
	email VARCHAR(50),
	cpf VARCHAR(10),
	endereco VARCHAR(50),
	município VARCHAR(50),
	uf VARCHAR(2),
	celular VARCHAR(50)
);

CREATE TABLE curso(
	codigo INTEGER PRIMARY KEY,
	nome VARCHAR(50),
	campus VARCHAR(50),
	turno VARCHAR(15)
);

CREATE TABLE disciplina(
	código INTEGER PRIMARY KEY,
	fk_rgm_aluno INTEGER FOREIGN KEY REFERENCES aluno(rgm),
	fk_codigo_curso INTEGER FOREIGN KEY REFERENCES curso(codigo),
	nome VARCHAR(50),
	faltas INTEGER,
	nota FLOAT,
	semestre VARCHAR(10)
);
